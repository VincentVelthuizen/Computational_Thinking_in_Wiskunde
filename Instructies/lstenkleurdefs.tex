\def\pijl{
	\begin{tikzpicture}[very thick,x=1em,y=1em,baseline={(0,-0.35em)}]
		\draw [->](-0.4,0) -- (0.4,0);
	\end{tikzpicture}
	}
\def\eindpijl{
	\begin{tikzpicture}[very thick,x=1em,y=1em,baseline={(0,-0.3em)}]
		\draw[<-] (-0.25,0) -- (0.25,0) -- (0.25,0.3); 
	\end{tikzpicture}
	}
